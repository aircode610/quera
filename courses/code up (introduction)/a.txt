دستگاه کمکی (ساده)
#include <bits/stdc++.h>

using namespace std;

int main(){
    int n, d, k;
    cin >> n >> d >> k;
    int digits = 0;
    while (n != 0){
        int dig = n%10;
        if (dig != d)
            digits++;
        n /= 10;
    }
    cout << digits*k << endl;
    return 0;
}

گُلمیناتور (واجب)
#include <bits/stdc++.h>

using namespace std;

int main() {
    int n, a;
    cin >> n >> a;
    int ans = 0, t = 1;
    while (n > 0){
        n -= t;
        t *= a;
        ans++;
    }
    cout << ans << endl;
    return 0;
}

غول مرحله آخر!
#include <bits/stdc++.h>

using namespace std;

int main() {
    int a, b;
    char op;
    cin >> a >> op >> b;
    if (op == '+')
        cout << a + b << endl;
    else if (op == '-')
        cout << a - b << endl;
    else if (op == '*')
        cout << a * b << endl;
    else if (op == '/')
        cout << a / b << endl;
    else if (op == 'P'){
        int ans = 1;
        for (int i = 1; i <= b; i++)
            ans *= a;
        cout << ans << endl;
    }
    else if (op == 'G'){
        int ans = 1;
        for (int i = 1; i <= a; i++)
            if (a % i == 0 && b % i == 0)
                ans = i;
        cout << ans << endl;
    }
    else {
        int sm = min(a, b);
        for (int i = 1; i <= max(a,b); i++){
            if (i*sm%max(a, b) == 0){
                cout << i*sm << endl;
                return 0;
            }
        }
    }
}
